### Task 1: Create Basiq API client (`src/lib/basiq.ts`)

Create a typed Basiq API client with four functions:

```typescript
// src/lib/basiq.ts

const BASIQ_BASE = "https://au-api.basiq.io";

/**
 * Fetch a SERVER_ACCESS token from Basiq.
 * Header: Authorization: Basic <base64(API_KEY + ":")>
 *         basiq-version: 3.0
 * Body: scope=SERVER_ACCESS (form-encoded)
 */
export async function getServerToken(apiKey: string): Promise<string>

/**
 * Create a Basiq user. Returns the Basiq userId.
 */
export async function createBasiqUser(
  token: string,
  email: string
): Promise<string>

/**
 * Create an auth link for a Basiq user. Returns the consent URL.
 */
export async function createAuthLink(
  token: string,
  basiqUserId: string
): Promise<string>

/**
 * Fetch all accounts for a Basiq user.
 */
export async function getBasiqAccounts(
  token: string,
  basiqUserId: string
): Promise<BasiqAccount[]>

export interface BasiqAccount {
  id: string;
  name: string;
  currency: string;
  balance: string;
  class: { type: string; product: string };
  status: string;
}
```

Implementation notes:
- `getServerToken`: POST `${BASIQ_BASE}/token`, Content-Type `application/x-www-form-urlencoded`,
  Authorization `Basic ${Buffer.from(apiKey + ":").toString("base64")}`,
  basiq-version `3.0`, body `scope=SERVER_ACCESS`
- `createBasiqUser`: POST `${BASIQ_BASE}/users`, Authorization `Bearer ${token}`,
  basiq-version `3.0`, body `{ email }`
- `createAuthLink`: POST `${BASIQ_BASE}/users/${basiqUserId}/auth_link`,
  Authorization `Bearer ${token}`, basiq-version `3.0`, body `{}`
  Returns `data.links.public`
- `getBasiqAccounts`: GET `${BASIQ_BASE}/users/${basiqUserId}/accounts`,
  Authorization `Bearer ${token}`, basiq-version `3.0`
  Returns `data.data` array (Basiq wraps in `{ data: [...] }`)

### Task 2: Create `src/server/api/routers/basiq.ts`

```typescript
import { createTRPCRouter, protectedProcedure } from "@/server/api/trpc";
import { env } from "@/env";
import {
  getServerToken,
  createBasiqUser,
  createAuthLink,
  getBasiqAccounts,
} from "@/lib/basiq";

export const basiqRouter = createTRPCRouter({
  /**
   * createConsentUrl
   * - Get server access token
   * - Find or create BasiqConnection for this user
   * - Create auth link, return the consent URL
   */
  createConsentUrl: protectedProcedure.mutation(async ({ ctx }) => {
    const userId = ctx.session.user.id;
    const token = await getServerToken(env.BASIQ_API_KEY);

    // Find existing connection or create a new Basiq user
    let connection = await ctx.db.basiqConnection.findFirst({
      where: { userId },
    });

    if (!connection) {
      const userEmail = ctx.session.user.email!;
      const basiqUserId = await createBasiqUser(token, userEmail);
      connection = await ctx.db.basiqConnection.create({
        data: { userId, basiqUserId },
      });
    }

    const consentUrl = await createAuthLink(token, connection.basiqUserId);
    return { consentUrl };
  }),

  /**
   * syncAccounts
   * - Get server access token
   * - Fetch accounts from Basiq
   * - Upsert BankAccount records
   * Returns list of synced accounts
   */
  syncAccounts: protectedProcedure.mutation(async ({ ctx }) => {
    const userId = ctx.session.user.id;

    const connection = await ctx.db.basiqConnection.findFirst({
      where: { userId },
    });

    if (!connection) {
      throw new TRPCError({
        code: "NOT_FOUND",
        message: "No bank connection found. Connect an account first.",
      });
    }

    const token = await getServerToken(env.BASIQ_API_KEY);
    const accounts = await getBasiqAccounts(token, connection.basiqUserId);

    // Upsert each account
    const upserted = await Promise.all(
      accounts.map((acc) =>
        ctx.db.bankAccount.upsert({
          where: { basiqAccountId: acc.id },
          create: {
            connectionId: connection.id,
            basiqAccountId: acc.id,
            name: acc.name,
            type: acc.class.type,
            balance: acc.balance,
            currency: acc.currency,
          },
          update: {
            name: acc.name,
            type: acc.class.type,
            balance: acc.balance,
          },
        })
      )
    );

    return upserted;
  }),
});
```

Add `import { TRPCError } from "@trpc/server";` at top.

<verify>
- [ ] createConsentUrl mutation creates BasiqConnection row on first call
- [ ] syncAccounts throws NOT_FOUND when no connection
- [ ] syncAccounts upserts by basiqAccountId
</verify>

### Task 3: Create `src/server/api/routers/account.ts`

```typescript
import { createTRPCRouter, protectedProcedure } from "@/server/api/trpc";

export const accountRouter = createTRPCRouter({
  list: protectedProcedure.query(async ({ ctx }) => {
    const userId = ctx.session.user.id;
    return ctx.db.bankAccount.findMany({
      where: {
        connection: { userId },
      },
      orderBy: { name: "asc" },
    });
  }),
});
```

### Task 5: Update `src/app/settings/accounts/page.tsx`

Replace placeholder with a full accounts page. This needs a client component wrapper
for interactivity plus server-rendered account list.

Create `src/app/settings/accounts/_components/AccountsClient.tsx` as a `"use client"` component
that:
- Lists connected accounts (from `api.account.list.useQuery()`)
- Shows "Connect bank account" button → calls `api.basiq.createConsentUrl.useMutation()`,
  then opens `consentUrl` in a new tab
- Shows "Sync accounts" button → calls `api.basiq.syncAccounts.useMutation()`,
  then invalidates `account.list`
- Shows loading/error states

Update `src/app/settings/accounts/page.tsx` to:
- Import `AccountsClient`
- Wrap in a `<main>` with heading

The `AccountsClient` component:
```tsx
"use client";

import { api } from "@/trpc/react";

export function AccountsClient() {
  const utils = api.useUtils();
  const { data: accounts, isLoading } = api.account.list.useQuery();

  const connectMutation = api.basiq.createConsentUrl.useMutation({
    onSuccess: ({ consentUrl }) => {
      window.open(consentUrl, "_blank");
    },
  });

  const syncMutation = api.basiq.syncAccounts.useMutation({
    onSuccess: () => {
      void utils.account.list.invalidate();
    },
  });

  return (
    <div className="mt-6 space-y-6">
      <div className="flex gap-3">
        <button
          onClick={() => connectMutation.mutate()}
          disabled={connectMutation.isPending}
          className="rounded-md bg-blue-600 px-4 py-2 text-sm font-medium text-white hover:bg-blue-700 disabled:opacity-50"
        >
          {connectMutation.isPending ? "Connecting..." : "Connect bank account"}
        </button>
        <button
          onClick={() => syncMutation.mutate()}
          disabled={syncMutation.isPending}
          className="rounded-md border border-gray-300 px-4 py-2 text-sm font-medium hover:bg-gray-50 disabled:opacity-50"
        >
          {syncMutation.isPending ? "Syncing..." : "Sync accounts"}
        </button>
      </div>

      {connectMutation.error && (
        <p className="text-sm text-red-600">{connectMutation.error.message}</p>
      )}
      {syncMutation.error && (
        <p className="text-sm text-red-600">{syncMutation.error.message}</p>
      )}

      {isLoading && <p className="text-sm text-gray-500">Loading accounts...</p>}

      {accounts && accounts.length === 0 && (
        <p className="text-sm text-gray-500">
          No accounts connected yet. Click "Connect bank account" to get started.
        </p>
      )}

      {accounts && accounts.length > 0 && (
        <ul className="divide-y divide-gray-200 rounded-lg border">
          {accounts.map((acc) => (
            <li key={acc.id} className="flex items-center justify-between p-4">
              <div>
                <p className="font-medium">{acc.name}</p>
                <p className="text-sm capitalize text-gray-500">{acc.type}</p>
              </div>
              <div className="text-right">
                <p className="font-medium">
                  {acc.currency} {Number(acc.balance).toFixed(2)}
                </p>
              </div>
            </li>
          ))}
        </ul>
      )}
    </div>
  );
}
```

<verify>
- [ ] Page renders without error
- [ ] Connect button calls createConsentUrl and opens URL in new tab
- [ ] Sync button calls syncAccounts and refreshes the list
- [ ] Account list shows name, type, balance
</verify>
